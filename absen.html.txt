  <h2>Absensi Pegawai</h2>
  <div id="nama"></div>
  <div>
    <button onclick="kirim('Hadir')">Hadir</button>
    <button onclick="kirim('Sakit')">Sakit</button>
    <button onclick="kirim('Tanpa Keterangan')">Tanpa Keterangan</button>
  </div>
  <p id="status"></p>

  <script>
    const params = new URLSearchParams(window.location.search);
    const nama = decodeURIComponent(params.get("nama") || "");

    document.getElementById("nama").innerHTML = `<strong>${nama}</strong>`;

    function kirim(status) {
      fetch("https://script.google.com/macros/s/AKfycbxIt9VWSzQ2p3D7xmE6Tp9ZFKG2lZWTNjhsm6RXSZWxVutH_J-BtYQ-4Y0RJRsE-p1ORw/exec", {
        method: "POST",
        body: JSON.stringify({ nama, status })
      })
      .then(res => res.json())
      .then(data => {
        document.getElementById("status").textContent = "✅ Absensi berhasil dikirim.";
      })
      .catch(err => {
        document.getElementById("status").textContent = "❌ Gagal mengirim absensi.";
      });
    }
  </script>
